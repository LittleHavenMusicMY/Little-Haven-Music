    <script>
        const starButtons = document.querySelectorAll('.star-rating button');
        let selectedRating = 0;

        starButtons.forEach(button => {
            button.addEventListener('click', () => {
                selectedRating = button.dataset.index;
                updateStars(selectedRating);
            });
        });

        function updateStars(rating) {
            starButtons.forEach(btn => {
                if (btn.dataset.index <= rating) {
                    btn.classList.add('text-amber-400');
                    btn.classList.remove('text-slate-100');
                } else {
                    btn.classList.remove('text-amber-400');
                    btn.classList.add('text-slate-100');
                }
            });
        }

        document.getElementById('reviewForm').addEventListener('submit', (e) => {
            e.preventDefault();
            if (selectedRating === 0) {
                alert("Please provide a rating for your testimonial.");
                return;
            }
            alert("✨ Thank you for sharing your story! Your testimonial has been submitted for verification.");
            e.target.reset();
            updateStars(0);
            selectedRating = 0;
        });

        document.getElementById('contactForm').addEventListener('submit', (e) => {
            e.preventDefault();
            alert("✨ Inquiry sent! Little Haven Music will contact you within 24 hours.");
            e.target.reset();
        });
    </script>
